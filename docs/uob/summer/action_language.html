  • holds(fluent, T) IF not occurs(action, T), holds(fluent, T-1), actions causes fluent change

Causal Laws

  • open(robot, loc_close) CAUSES opened(loc_close)
  • close(robot, loc_close) CAUSES not opened(loc_close)
  • pick(robot, object) CAUSES holding(robot, object), not in(object, location)
  • place(robot, object, location) CAUSES not holding(robot, object)
  • navigate(robot, location) CAUSES at(robot, location), explored(location)

%% instance.lp

% objects
#books = [book][1..2].
#food = {apple, orange}.
#object = #books + #food.

#space_open = {table, sofa}.
#container_close = {drawer}.
#container_open = {box}.
#container = #container_close + #container_open.
% space that needed 'open' action before operating
#loc_open =  #space_open + #container_open.
% space that dose NOT needed 'open' action before operating
#loc_close = #container_close.
#location = #loc_open + #loc_close.

% agents
#robot = {rob1}.


%% actions %%%%%%%%%%%%%%%%%%%%%%%%%%
#action =
% open drawer/fridge/door etc.
open(#robot, #loc_close)+
% opposite of open
close(#robot, #loc_close)+
% pick up objects
pick(#robot, #object) +
% opposite of pick
place(#robot, #object, #location) +
% TODO: is observe needed? How to refresh current scene else?
observe(#robot, #location, #object) +
% go to location from current place
navigate(#robot, #location).


%% fluents: inertial %%%%%%%%%%%%%%%%%%%%%%%%%%
#inertial_fluent =
opened(#location) +
emptied(#location) +
% if object in/on location
in(#object, #location) +
% is robot at location
at(#robot, #location) +
% location has been visited
explored(#location) +
% is robot holding the object
holding(#robot, #object).


%% fluents: defined %%%%%%%%%%%%%%%%%%%%%%%%%%
% #defined_fluent

#fluent = #inertial_fluent. % + #defined_fluent.

%% history: %%%%%%%%%%%%%%%%%%%%%%%%%%
#step = 1..k.
holds(#fluent, T).
occurs(#action, T).
